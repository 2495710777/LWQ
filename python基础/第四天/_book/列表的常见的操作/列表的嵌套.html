列表的嵌套

操作嵌套列表，只要把要操作元素的下标当作变量名来使用即可。

a = ['hello', 'hi', 'good']
b = [[34, 23, 200, 56], ['hello', 'hi', 'xixixi', 'heiheihei'], ['True', 'False']]
print(b[1][1])

for i, x in enumerate(b):
    print('第%d个元素有%d个数据，是' % (i, len(x)), end='')
    for y in x:
        print(y, end='\t')
    print()

嵌套练习

#  一个学校，有3个办公室，现在有8位老师等待工位的分配，请编写程序，完成随机的分配
import random
offices = [[], [], []]
teachers = ['A', 'B', 'C', 'D', 'E', 'F', 'G', 'H', 'I', 'J', 'K']
for teacher in teachers:
    # randint(a,b) 生成[a,b)的随机数
    i = random.randint(0, 2)
    offices[i].append(teacher)
print(offices)
for i, office in enumerate(offices):
    print('第%d个房间里有%d个人:' % (i, len(office)), end='')
    for teacher in office:
        print(teacher, end='')
    print()
