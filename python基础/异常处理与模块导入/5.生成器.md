```python
# 生成器就是一个普通的函数，就是return换成yield
def my_generator(n):
    current = 0
    while current < n:
        yield current  # 下次接着走，所以可以放这里
        current += 1
gen = my_generator(5)
print(gen)
# print(next(gen))
# # 上次从哪儿(yield)停，下次从哪儿(yield)进
# print(next(gen))
# print(next(gen))
# print(next(gen))
# print(next(gen))
for g in gen:
    print(g)
```

### send的使用

```python
def my_generator(n):
    current = 0
    while current < n:
        temp = yield current
        print(temp)
        current += 1


gen = my_generator(5)
print(gen)
print(next(gen))  # 0


# 上次从哪儿(yield)停，下次从哪儿(yield)进
# send 把haha传给yield current赋值给temp
print(gen.send('haha'))  # haha,0
```

### 生成器生成菲波那切数列

```python
def fib_generator(n):
    current = 0
    num1 = num2 = 1
    while current < n:
        current += 1
        yield num1
        num1, num2 = num2, num1 + num2


gen = fib_generator(12)
# print(next(gen))
# print(next(gen))
# print(next(gen))
# print(next(gen))
# print(next(gen))
for g in gen:
    print(g)

```
