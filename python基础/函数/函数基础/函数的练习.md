# 写一个函数打印一个横线
def print_line():
    print('-' * 30)
print_line()
# 写一个函数打印一个指定行数的横线
def print_lines(n):
    i = 0
    while i < n:
        i += 1
        print('-' * 30)
print_lines(5)
# 写一个函数求三个数的和
def sum3num(a, b, c):
    return a + b + c
print(sum3num(5, 1, 9))
# 写一个函数求三个数的平均值
def ave3num(a, b, c):  # a = 5, b = 7, c = 3
    return sum3num(a, b, c) / 3  #
print(ave3num(5, 7, 3))
# 写一个函数求任意个数数字之和
def get_sum(a):  # a = (1, 5, 3, 7)
    result = 0
    for i in a:
        result += i
    return result
m = (1, 5, 3, 7)
print(get_sum(m))
