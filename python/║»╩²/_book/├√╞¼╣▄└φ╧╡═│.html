名片管理系统

import time

list1 = [{'name': 'zhangsan', 'tel': '123', 'qq': '321'},
         {'name': 'lisi', 'tel': '123', 'qq': '321'},
         {'name': 'wangwu', 'tel': '123', 'qq': '321'},
         {'name': 'tommy', 'tel': '123', 'qq': '321'}]


def add_card():
    new_name = input('请输入姓名：')
    for inname in list1:
        if inname['name'] == new_name:
            print('此名已被占用，请重新输入')
            add_card()
    new_tel = input('请输入电话：')
    new_qq = input('请输入qq')
    card = {'name': new_name, 'tel': new_tel, 'qq': new_qq}
    list1.append(card)
    time.sleep(0.5)
    print('添加成功')
    is_sure = input('是否继续添加？(yes or no):')
    if is_sure.lower() == 'yes':
        add_card()


def del_card():
    num = input('请输入您要删除的序号：')
    if not num.isdigit():
        print('输入有误，请输入整数')
        return
    if 0 <= int(num) < len(list1):
        is_sure = input('您确定要删除么？(yes or no):')
        if is_sure.lower() == 'yes':
            del list1[int(num)]
    else:
        print('输入有误，请重新输入')
    time.sleep(1)


def modify_card():
    num = input('请输入您要修改的序号：')
    if not num.isdigit():
        print('输入有误，请输入整数')
        return
    if not 0 <= int(num) < len(list1):
        print('输入有误，请重新输入')
        return
    user = list1[int(num)]
    in_fo = ''
    for i, v in user.items():
        in_fo += '{}:{} '.format(i, v)
    print('您要修改的信息是：')
    print(in_fo)
    new_name = input('请输入新的姓名：')
    for inname in list1:
        if inname['name'] == new_name:
            print('此名已被占用，请重新输入')
            return
    new_tel = input('请输入新的电话：')
    new_qq = input('请输入新的qq：')
    user['name'] = new_name
    user['tel'] = new_tel
    user['qq'] = new_qq

    print('修改成功')
    time.sleep(0.5)


def search_card():
    search_name = input('请输入你要查找的姓名：')
    for card in list1:
        if card['name'] == search_name:
            print('您查找的信息是：')
            print('name:{name} tel:{tel} qq:{qq}'.format(**card))
            time.sleep(0.5)
            return
    print('您查找的信息不存在~~')
    time.sleep(0.5)
def show_card():
    print('序号\t姓名\t手机号\tqq')
    for i,v in enumerate(list1):
        print(i,v['name'].center(10),v['tel'].center(5),v['qq'],sep='\t')
    time.sleep(5)


def exit_card():
    is_sure = input('您确定要退出么？(yes or no):')
    if is_sure.lower() == 'yes':
        return True
    return False


def main():
    while True:
        print("---------------------------")
        print("      名片管理系统 V1.0")
        print(" 1:添加名片".center(20))
        print(" 2:删除名片".center(20))
        print(" 3:修改名片".center(20))
        print(" 4:查询名片".center(20))
        print(" 5:显示所有名片".center(20))
        print(" 6:退出系统".center(20))
        print("---------------------------")
        num = input('请输入要进行的操作（数字）:')
        if num == '1':
            add_card()
        elif num == '2':
            del_card()
        elif num == '3':
            modify_card()
        elif num == '4':
            search_card()
        elif num == '5':
            show_card()
        elif num == '6':
            is_True = exit_card()
            if is_True:
                break
        else:
            print('输入有误，请重新输入')


main()
